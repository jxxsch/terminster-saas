Terminster Booking Widget

Version 2.0 - Multi-tenant, Multi-language, Light/Dark Theme

Dark Theme (Deutsch) data-theme="dark" data-locale="de"
Light Theme (English) data-theme="light" data-locale="en"
Turkce (Turkish) data-locale="tr"
Custom Color data-color="#FF6B6B"

Integration Guide

1. Simple HTML (Auto-Init)

<!-- Add this where you want the widget -->
<div
  data-terminster-widget
  data-auto-init
  data-tenant="your-tenant"
  data-shop="your-shop"
  data-theme="dark"
></div>

<!-- Add script at end of body -->
<script src="https://widget.terminster.de/booking-widget.umd.js"></script>

2. JavaScript Initialization

<div id="booking-widget"></div>

<script src="https://widget.terminster.de/booking-widget.umd.js"></script>
<script>
  const widget = TerminsterWidget.init({
    tenant_slug: 'your-tenant',
    shop_slug: 'your-shop',
    theme: 'dark',
    locale: 'de',
    primary_color: '#D4AF37',
    on_success: (appointment) => {
      console.log('Booked!', appointment);
    },
    on_error: (error) => {
      console.error('Error:', error);
    }
  });

  // Later: widget.destroy() or widget.update({ theme: 'light' })
</script>

3. Configuration Options

interface WidgetConfig {
  tenant_slug: string;       // Required: Your tenant identifier
  shop_slug?: string;        // Optional: Specific shop (or show selector)
  api_url?: string;          // Optional: Custom API URL
  theme?: 'light' | 'dark' | 'auto';  // Default: 'dark'
  primary_color?: string;    // Optional: Brand color (hex)
  locale?: 'de' | 'en' | 'tr';  // Default: 'de'
  show_header?: boolean;     // Show title + progress (default: true)
  show_shop_selector?: boolean;  // Multi-location picker (default: true)
  allow_any_barber?: boolean;  // "Any barber" option (default: true)
  on_success?: (appointment) => void;  // Success callback
  on_error?: (error) => void;  // Error callback
}

4. Data Attributes

data-terminster-widget  // Required: Marks element as widget container
data-auto-init          // Auto-initialize on page load
data-tenant="slug"      // Tenant identifier
data-shop="slug"        // Shop identifier (optional)
data-theme="dark"       // Theme: light, dark, auto
data-locale="de"        // Language: de, en, tr
data-color="#D4AF37"    // Primary color
data-header="false"     // Hide header
data-shop-selector="false"  // Hide shop selector
data-any-barber="false"  // Hide "any barber" option
data-api-url="..."      // Custom API URL

5. API Methods

// Global API
TerminsterWidget.init(config)      // Initialize on default container
TerminsterWidget.initOn(el, config)  // Initialize on specific element
TerminsterWidget.destroyAll()      // Destroy all widget instances
TerminsterWidget.version           // Current version string

// Instance methods (returned by init/initOn)
widget.destroy()                   // Remove widget
widget.update(config)              // Update configuration
